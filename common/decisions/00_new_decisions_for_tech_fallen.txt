move_tech_fallen_key_here = {
	owned_planets_only = yes
	icon = decision_consecrate_habitat_knights

	enactment_time = 30
	resources = {
		category = decisions
		cost = {
			unity = 500
		}
	}

	potential = {
		if = {
			limit = {
				owner = {
					is_ai = yes
				}
			}
			has_building = building_master_archive
		}
		owner = {
			any_controlled_planet = {
				has_modifier = planet_has_tech_fallen_key
			}
		}
	}

	allow = {
		custom_tooltip = {
			fail_text = "tech_fallen_key_already_be_here"
			NOT = {
				has_modifier = planet_has_tech_fallen_key
			}
		}
	}

	effect = {
		owner = {
			every_owned_planet = {
				limit = {
					has_modifier = planet_has_tech_fallen_key
				}
				remove_modifier = planet_has_tech_fallen_key
			}
		}
		if = {
			limit = {
				OR = {
					has_building = building_master_archive
					has_building = building_master_archive2
				}
			}
			hidden_effect = {
				planet_event = {
					id = tech_fallen_events.27
				}
			}
		}
		add_modifier = {
			modifier = planet_has_tech_fallen_key
		}
	}

	ai_weight = {
		weight = 99999
	}
}
